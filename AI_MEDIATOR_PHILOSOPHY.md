## 🎯 Design Principles

### 1. **Invisible Interface**
The best interface is **no interface**. The UI should vanish when you don't need it.

```yaml
Default State:
  - Fullscreen transparent overlay
  - Single input field (voice or text)
  - No menus, no buttons, no chrome
  - Appears on demand (hotkey, gesture, voice)
  
Active State:
  - Minimal visual feedback
  - Results appear contextually
  - Disappears after task completion
```

### 2. **Natural Interaction**
Interaction should be as natural as breathing:

```yaml
Input Methods:
  Voice:
    - "Show me sales data from last quarter"
    - "What's the weather in Tokyo?"
    - "Find research on quantum computing"
  
  Gesture:
    - Swipe down from anywhere → instant search
    - Draw circle → contextual menu

  Thought (Future):

    - Predictive pre-loading
```

### Flutter UI Redesign

```dart
// Main screen: Invisible by default
class AirInterface extends StatefulWidget {
  @override
  _AirInterfaceState createState() => _AirInterfaceState();
}

class _AirInterfaceState extends State<AirInterface> 
    with SingleTickerProviderStateMixin {
  
  late AnimationController _breathController;
  bool _isVisible = false;
  
  @override
  void initState() {
    super.initState();
    _breathController = AnimationController(
      duration: Duration(seconds: 4),
      vsync: this,
    );
    
    // Listen for hotkey (Cmd+Space)
    HardwareKeyboard.instance.addHandler(_handleKeyPress);
    
    // Listen for voice activation
    VoiceService.instance.onWakeWord('hey zero', () {
      _summon();
    });
  }
  
  bool _handleKeyPress(KeyEvent event) {
    if (event is KeyDownEvent) {
      if (event.logicalKey == LogicalKeyboardKey.space && 
          HardwareKeyboard.instance.isMetaPressed) {
        _summon();
        return true;
      }
    }
    return false;
  }
  
  void _summon() {
    setState(() => _isVisible = true);
    _breathController.forward(); // Inhale animation
  }
  
  void _dismiss() {
    _breathController.reverse().then((_) {
      setState(() => _isVisible = false);
    });
  }
  
  @override
  Widget build(BuildContext context) {
    return AnimatedOpacity(
      opacity: _isVisible ? 1.0 : 0.0,
      duration: Duration(milliseconds: 300),
      curve: Curves.easeInOutSine,
      child: Container(
        color: Color(0xF5000000), // Almost transparent black
        child: Center(
          child: ScaleTransition(
            scale: Tween<double>(begin: 0.8, end: 1.0).animate(
              CurvedAnimation(
                parent: _breathController,
                curve: Curves.easeOutCubic,
              ),
            ),
            child: _buildInputField(),
          ),
        ),
      ),
    );
  }
  
  Widget _buildInputField() {
    return Container(
      width: 600,
      padding: EdgeInsets.symmetric(horizontal: 24, vertical: 16),
      decoration: BoxDecoration(
        color: Color(0x1A87CEEB), // Sky blue, very transparent
        borderRadius: BorderRadius.circular(30),
        border: Border.all(
          color: Color(0x5500CED1), // Breeze cyan, semi-transparent
          width: 1,
        ),
        boxShadow: [
          BoxShadow(
            color: Color(0x3300CED1), // Glow effect
            blurRadius: 20,
            spreadRadius: 2,
          ),
        ],
      ),
      child: Row(
        children: [
          Icon(
            Icons.air, // Air icon
            color: Color(0xFF87CEEB),
            size: 24,
          ),
          SizedBox(width: 12),
          Expanded(
            child: TextField(
              autofocus: true,
              style: TextStyle(
                color: Color(0xFFE8E8E8),
                fontSize: 18,
                fontWeight: FontWeight.w300,
              ),
              decoration: InputDecoration(
                hintText: 'Ask anything... (or say "Hey Zero")',
                hintStyle: TextStyle(
                  color: Color(0xFF888888),
                ),
                border: InputBorder.none,
              ),
              onSubmitted: (query) => _handleQuery(query),
            ),
          ),
          IconButton(
            icon: Icon(
              Icons.mic,
              color: Color(0xFF00CED1),
            ),
            onPressed: () => _startVoiceInput(),
          ),
        ],
      ),
    );
  }
  
  Future<void> _handleQuery(String query) async {
    // Show breathing animation
    _breathController.repeat(reverse: true);
    
    // Send to AI mediator
    final response = await AIMediator.instance.mediate(query);
    
    // Stop breathing, show result
    _breathController.stop();
    _showResult(response);
  }
  
  void _showResult(AIResponse response) {
    showDialog(
      context: context,
      builder: (context) => BreathingResultDialog(response: response),
    );
    
    // Auto-dismiss after 5 seconds
    Future.delayed(Duration(seconds: 5), () {
      if (mounted) Navigator.of(context).pop();
      _dismiss();
    });
  }
}
```
